rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Users koleksiyonu
    match /users/{userId} {
      // Herkes kendi profilini okuyup yazabilir
      allow read, write: if request.auth != null && request.auth.uid == userId;
      
      // Email ile arama için - tüm giriş yapmış kullanıcılar okuyabilir
      allow read: if request.auth != null;
    }
    
    // Projects koleksiyonu (root seviyede)
    match /projects/{projectId} {
      // Proje oluşturma - giriş yapmış herkes
      allow create: if request.auth != null;
      
      // Okuma - proje sahibi veya ekip üyesi
      allow read: if request.auth != null && 
                     (request.auth.uid == resource.data.ownerId ||
                      request.auth.uid in resource.data.teamMemberIds);
      
      // Güncelleme - proje sahibi veya ekip üyesi
      allow update: if request.auth != null && 
                       (request.auth.uid == resource.data.ownerId || 
                        request.auth.uid in resource.data.teamMemberIds);
      
      // Silme - sadece proje sahibi
      allow delete: if request.auth != null && 
                       request.auth.uid == resource.data.ownerId;
      
      // Tasks sub-koleksiyonu
      match /tasks/{taskId} {
        allow read, write: if request.auth != null && 
                              (request.auth.uid == get(/databases/$(database)/documents/projects/$(projectId)).data.ownerId || 
                               request.auth.uid in get(/databases/$(database)/documents/projects/$(projectId)).data.teamMemberIds);
      }
    }
  }
}
